        // Create Dynamic Background Elements - Only Stars and Meteors
        function createDynamicBackground() {
            const dynamicBg = document.getElementById('dynamic-bg');
            
            // Create three layers of stars
            createStarsLayer('stars-layer-1', 100, 1, 3, 'twinkle-slow');
            createStarsLayer('stars-layer-2', 150, 2, 3, 'twinkle-medium');
            createStarsLayer('stars-layer-3', 200, 3, 3, 'twinkle-fast');

        // Create Star Layers
        function createStarsLayer(layerId, count, sizeMultiplier, speedMultiplier, animationType) {
            const layer = document.getElementById(layerId);
            
            for (let i = 0; i < count; i++) {
                const star = document.createElement('div');
                star.classList.add('star');
                
                // Random size
                const size = Math.random() * 3 * sizeMultiplier;
                star.style.width = `${size}px`;
                star.style.height = `${size}px`;
                
                // Random position
                star.style.left = `${Math.random() * 100}%`;
                star.style.top = `${Math.random() * 100}%`;
                
                // Random twinkle duration
                const twinkleDuration = Math.random() * 3 * speedMultiplier + 2;
                star.style.setProperty('--twinkle-duration', `${twinkleDuration}s`);
                
                // Random delay
                star.style.animationDelay = `${Math.random() * 5}s`;
                
                // Random brightness
                const brightness = Math.random() * 0.7 + 0.3;
                star.style.opacity = brightness;
                
                layer.appendChild(star);
            }
        }
        
        // Create Meteors
        function createMeteors(count) {
            const dynamicBg = document.getElementById('dynamic-bg');
            
            for (let i = 0; i < count; i++) {
                const meteor = document.createElement('div');
                meteor.classList.add('meteor');
                
                // Randomly select meteor type (1-5)
                const meteorType = Math.floor(Math.random() * 5) + 1;
                meteor.classList.add(`type-${meteorType}`);
                
                // Random position - starting from the top
                meteor.style.left = `${Math.random() * 100}%`;
                meteor.style.top = `-${Math.random() * 300 + 100}px`;
                
                // Random duration (2-6 seconds)
                const meteorDuration = Math.random() * 4 + 2;
                meteor.style.setProperty('--meteor-duration', `${meteorDuration}s`);
                
                // Random delay (0-30 seconds)
                meteor.style.animationDelay = `${Math.random() * 30}s`;
                
                // Random color brightness
                const brightness = Math.random() * 0.8 + 0.2;
                meteor.style.opacity = brightness;
                
                // Random rotation angle variation
                const rotation = -45 + (Math.random() * 10 - 5);
                meteor.style.transform = `rotate(${rotation}deg)`;
                
                dynamicBg.appendChild(meteor);
                
                // Add explosion effect to some meteors
                if (Math.random() > 0.7) {
                    setTimeout(() => {
                        createMeteorExplosion(meteor);
                    }, meteorDuration * 1000 * 0.8);
                }
            }
        }
        
        // Create Meteor Explosion Effect
        function createMeteorExplosion(meteor) {
            const explosion = document.createElement('div');
            explosion.classList.add('meteor-explosion');
            
            // Get meteor position
            const meteorRect = meteor.getBoundingClientRect();
            explosion.style.left = `${meteorRect.left + meteorRect.width/2}px`;
            explosion.style.top = `${meteorRect.top + meteorRect.height/2}px`;
            
            // Random explosion color
            const colors = [
                'white',
                'var(--tech-blue)',
                'var(--eco-green)',
                'rgba(255, 255, 255, 0.8)'
            ];
            const randomColor = colors[Math.floor(Math.random() * colors.length)];
            explosion.style.backgroundColor = randomColor;
            
            document.body.appendChild(explosion);
            
            // Remove element after explosion
            setTimeout(() => {
                if (explosion.parentNode) {
                    explosion.parentNode.removeChild(explosion);
                }
            }, 1000);
        }
        
        // Drone Auto Transformation
        function startAutoTransform() {
            const droneH = document.querySelector('.drone-h');
            const droneI = document.querySelector('.drone-i');
            const dotH = document.getElementById('dot-h');
            const dotI = document.getElementById('dot-i');
            
            let isHType = true;
            
            // Auto switch every 5 seconds
            setInterval(() => {
                if (isHType) {
                    // Switch to I-type
                    droneH.style.opacity = '0';
                    droneI.style.opacity = '1';
                    dotH.classList.remove('active');
                    dotI.classList.add('active');
                } else {
                    // Switch to H-type
                    droneH.style.opacity = '1';
                    droneI.style.opacity = '0';
                    dotH.classList.add('active');
                    dotI.classList.remove('active');
                }
                
                isHType = !isHType;
            }, 5000);
        }
